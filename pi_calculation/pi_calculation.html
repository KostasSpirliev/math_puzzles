        xx <- center[1] + r * cos(tt)
        yy <- center[2] + r * sin(tt)
        return(data.frame(x = xx, y = yy))

calc_pi <- function(n) {
    x <- runif(n, min = -1, max = 1)
    y <- runif(n, min = -1, max = 1)
    dist <- x^2 + y^2
    a <- ifelse(dist < 1, 1, 0)
    4*sum(a)/length(a)
}
